རྫོགས་ཆེན་ཤྲཱི་སིངྷའི་བཤད་གྲྭར་ཞུགས་ཤིང་མདོ་སྔགས་གཞུང་ལུགས་རབ་འབྱམས་ལ་བརྩོན་པ་ལྷོད་མེད་གནང་། མཁན་པོ་ལྷ་སྒང་མདུན་ནས་བསྙེན་པར་རྫོགས། ཞེ་ཆེན་དགོན་དུ་ཟླ་བ་དྲུག་བཞུགས་རིང་ཞེ་ཆེན་ཀོང་སྤྲུལ་རིན་པོ་ཆེའི་མདུན་ནས་རྫོགས་པ་ཆེན་པོ་རང་སྣང་རིས་མེད་ཀྱི་རིག་པ་ངོ་སྤྲོད་ཞུས་པས་རྟོགས་པའི་གོ་འཕང་མཐོན་པོར་གཤེགས། ༡༩༥༧ ལོར་ཀུན་སྤང་བྱ་བྲལ་བའི་ཚུལ་གྱིས་ཡར་ལུང་རྟེན་གསུམ་གནས་གསུམ་སོགས་ཨོ་རྒྱན་ཆེན་པོའི་སྒྲུབ་གནས་ཡོངས་རྫོགས་མཇལ་ཞིང་སྐུ་མཚམས་བཞུགས། ༡༩༥༩ ལོར་ཡོལ་མོ་གངས་ར་བརྒྱུད་བཙན་བྱོལ་དུ་ཕེབས། ༡༩༦༠ ལོར་རྒྱ་གར་ཀ་ལེན་སྤུངས་སུ་༸སྐྱབས་རྗེ་མཁྱེན་བརྩེ་རིན་པོ་ཆེ་དང་། ༸སྐྱབས་རྗེ་རྡོ་གྲུབ་རིན་པོ་ཆེ་རྣམ་གཉིས་ཀྱི་སྐུ་མདུན་ནས་སྙིང་ཐིག་ཡ་བཞི། འཇིགས་གླིང་པོད་དགུ། འཇམ་མགོན་མི་ཕམ་པའི་གསུང་འབུམ་སོགས་གསན། ༡༩༦༢ ལོར་འབྲུག་གཞུང་གི་རིག་གནས་སློབ་གྲྭ་སེམས་ཏོག་ཁའི་དགེ་རྒན་གནང་སྟེ་ལོ་གཉིས་བཞུགས་ཤིང་བསྟན་བཅོས་ཀྱང་ཅི་རིགས་བརྩམས། ཌ་ལ་ཧོ་སི་སྤྲུལ་སྐུའི་སློབ་གྲྭའི་དགེ་རྒན་ལོ་ཤས་གནང་། ༡༩༦༧ ལོར་ངེས་དོན་དགའ་ཚལ་གླིང་གི་བློ་གསལ་རྣམས་ལ་དབུ་མ། ཕར་ཕྱིན། ཚད་མ། སྤྱོད་འཇུག་སོགས་དཔེ་ཁྲིད་གནང་སྟེ་ལོ་ལྔ་ལྷག་ཙམ་བཞུགས། དེ་སྐབས་སུམ་རྟགས་འགྲེལ་པ་དང་རྟགས་རིགས་སོགས་བརྩམས། དེ་ནས་ལོ་གཉིས་རིང་ཝར་ཎ་བོད་ཀྱི་ཆེས་མཐོའི་སློབ་གཉེར་ཁང་གི་དགེ་རྒན་གནང་། ༡༩༧༡ ནས་ ༡༩༧༧ བར་འབྲས་ལྗོངས་གཞུང་གི་བཤད་གྲྭ་ཐུབ་བསྟན་མདོ་སྔགས་ཆོས་གླིང་གི་མཁན་ཐོག་གསུམ་པའི་ཐུགས་འགན་བཞེས་ཤིང་མདོ་སྔགས་ཀྱི་གཞུང་ཁྲིད་གནང་། ༡༩༧༧ ཟླ་ ༡༡ ཚེས་ ༡༩ ཉིན་ ༸སྐྱབས་རྗེ་གྲུབ་དབང་པདྨ་ནོར་བུ་མཆོག་གིས་གདན་ཞུས་ལྟར་དགོན་འདིར་ཕེབས། ༡༩༧༨ བོད་ཟླ་ ༡ ཚེས་༡༠ ཉིན་མཐོ་སློབ་འདི་ཉིད་ཐོག་མར་དབུ་བཙུགས་པའི་རྟེན་འབྲེལ་སྐབས་ཐོས་པའི་ཕན་ཡོན་ནས་བརྩམས་ཏེ་བཤད་གྲྭའི་དཔེ་ཁྲིད་ཀྱི་དབུ་འཛུགས་གནང་བ་༸སྐྱབས་རྗེ་རིན་པོ་ཆེས་ཀྱང་གསན། གཞན་སློབ་ཕྲུག་བཅུ་བཞི་ཙམ་ལ་སྤྱོད་འཇུག་སྡོམ་གསུམ། སུམ་རྟགས། སྙན་ངག རྒྱལ་སྲས་ལག་ལེན། ཁྱད་པར་འཕགས་བསྟོད། རྟགས་རིགས་བཅས་དེ་སྐབས་ལོ་རིམ་དང་པོའི་སློབ་ཚན་ཆ་ཚང་ཁྲིད་གནང་བ་མ་ཟད་ལོ་དེའི་ཡིག་རྒྱུགས་དང་བཅས་པ་འཛིན་སྐྱོང་གནང་། ༸སྐྱབས་རྗེ་རིན་པོ་ཆེས་བཀས་བསྐུལ་བ་བཞིན། མཐོ་སློབ་འདིའི་སློབ་ཚན་ཆ་ཚང་གཏན་འབེབས་གནང་བ་ཕྱིས་སུ་ཅུང་ཟད་ལེགས་བཅོས་བྱས་པ་ལས་གཞན་རྣམས་ད་ལྟ་ཡང་ཕྱག་ལེན་བསྟར་བཞིན་པའོ། །དེ་ལྟར་ལོ་གཅིག་གི་མཛད་པ་ལེགས་པར་གྲུབ་རྗེས་ ༡༩༧༩ ལོར་ཀ་ལ་མན་གྲུར་༸སྐྱབས་རྗེ་མཁྱེན་བརྩེ་རིན་པོ་ཆེས་རིན་ཆེན་གཏེར་མཛོད་ཀྱི་དབང་དང་། ༸སྐྱབས་རྗེ་རྡོ་གྲུབ་རིན་པོ་ཆེས་ལུང་གསུངས་པ་རྣམས་གསན་སྐབས་དབང་ལུང་གྲུབ་ལ་ཉེ་བ་ཟླ་བོད་ ༡༢ ཚེས་ ༡༨ ཀུན་མཁྱེན་ཆོས་ཀྱི་རྒྱལ་པོའི་འདས་མཆོད་ཉིན། མངོན་རྟོགས་རྒྱན་གྱི་མཆོད་བརྗོད་སྐོར་རྫོགས་སྨིན་སྦྱངས་གསུམ་ལས་ཞིང་སྦྱོང་ཚུལ་ནས་འཕྲོས་བདེ་བ་ཅན་གྱི་ཞིང་བཀོད་བསྡུས་ཙམ་ཞིག་གསུང་བཞིན་རེ་ཞིག་དགོངས་པ་ཆོས་དབྱིངས་སུ་བསྡུས་པའི་ཚུལ་བསྟན་ཏོ།།

ཕྱིར་ལོག

༄༅། །མཁན་ཆེན་པདྨ་ཤེས་རབ་རིན་པོ་ཆེ་མཆོག་གི་མཛད་རྣམ།

སྤྱི་ལོ་ ༡༩༣༦ ལོར་ཁམས་སྡེ་དགེ་འཇོ་མདའི་ཁོངས་རི་ཕུ་ཞེས་པར་ཡབ་རྡོ་རྗེ་དོན་གྲུབ་དང་། ཡུམ་ཚེ་མཚོ་གཉིས་ཀྱི་སྲས་སུ་སྐུ་འཁྲུངས། དགུང་གྲངས་བརྒྱད་དགུ་ཙམ་ནས་ཕྱུགས་འཚོ་བའི་ཞོར་དུ་ཞང་པོ་བླ་མ་ཆོས་བཟང་ཞུ་བ་ལས་འབྲི་ཀློག་སོགས་ལེགས་པར་སྦྱངས། བཅུ་བཞི་བསྐོར་ནས་ལྷ་ལྡན་ཕྱོགས་སུ་ཕེབས་ཏེ་རིས་མེད་སྐྱེས་ཆེན་དུ་མ་མཉེས་པ་གསུམ་གྱིས་བསྟེན་ཅིང་ཉིན་མཚན་ལྟོས་མེད་ཀྱི་སྒོ་ནས་སློབ་གཉེར་མང་དུ་གནང་། ༡༩༥༣ ལོར་ཞེ་ཆེན་ཀོང་སྤྲུལ་རིན་པོ་ཆེའི་མདུན་ནས་ཚིགས་སྔ་མ་དགེ་ཚུལ་གྱི་སྡོམ་པ་བཞེས། ༡༩༥༩ ལོར་བཙན་བྱོལ་དུ་ཕེབས། ༸སྐྱབས་རྗེ་མཁྱེན་བརྩེ་རིན་པོ་ཆེའི་སྐུ་མདུན་ནས་རྩ་རྒྱུད་གསང་བ་སྙིང་པོ་དང་། གདམས་ངག་མཛོད། བཀའ་བརྒྱུད་སྔགས་མཛོད། གཏེར་མཛོད། ཡོན་ཏན་མཛོད། སྡོམ་གསུམ་དཔག་སྙེ། ཀུན་བཟང་བླ་མའི་ཞལ་ལུང་སོགས་མདོ་སྔགས་ཀྱི་ཆོས་མང་དུ་གསན། གཞན་ཡང་སྔ་ཕྱི་ཅི་རིགས་སུ་༸སྐྱབས་རྗེ་བདུད་འཇོམས་རིན་པོ་ཆེ་དང་། ༸སྐྱབས་རྗེ་གྲུབ་དབང་པདྨ་ནོར་བུ་རིན་པོ་ཆེ་རྣམ་གཉིས་ཀྱི་མདུན་ནས་རིན་ཆེན་གཏེར་མཛོད་ཆེན་མོའི་དབང་ལུང་སོགས་བཀའ་དབང་དུ་མ་དང་། ༸སྐྱབས་རྗེ་སྟག་ལུང་རྩེ་སྤྲུལ་རིན་པོ་ཆེའི་སྐུ་མདུན་ནས་བྱང་གཏེར་ཡོངས་རྫོགས་ཀྱི་དབང་ལུང་གསན། ༡༩༦༨ ལོར་ ༸སྐྱབས་རྗེ་གྲུབ་དབང་རིན་པོ་ཆེས་བཀས་བསྐུལ་བ་ལྟར་དགོན་འདིར་ཞབས་སོར་འཁོད། དེ་སྐབས་བཤད་གྲྭ་ཟུར་བཙུགས་མེད་ཀྱང་ལོ་ཆུང་བློ་གསལ་གྲྭ་བཙུན་རྣམས་ལ་རིག་གནས་སོགས་སློབ་ཁྲིད་གནང་སྟེ་ལོ་ཤས་བཞུགས། ༡༩༧༨ ལོར་མཐོ་སློབ་འདི་ཉིད་དབུ་བརྙེས་པ་ནས་བཟུང་ད་ལྟ་བར་རྙེད་བདེ་གྲགས་བསྟོད་སོགས་རྒྱབ་ཏུ་བོར་ཏེ། རྒྱལ་བསྟན་སྤྱི་དང་ཁྱད་པར་ཆེ་བ་དྲུག་ལྡན་གྱི་སྔ་འགྱུར་བའི་བསྟན་པ་ཁོ་ན་ཐུགས་ལ་བཞག་སྟེ་སྐུ་ངལ་འཛེམས་མེད་ཐོག་མཐོ་སློབ་ཆེན་པོ་འདིའི་མཚོན་པའི་དད་ལྡན་སློབ་གཉེར་བ་རྣམས་ལ། ཐུན་མོང་སུམ་རྟགས་དག་གསུམ་དང་སྙན་ངག ཐུན་མིན་གཞུང་ཆེན་བཅུ་གསུམ་གཙོས་ཡོན་ཏན་མཛོད། རྩ་རྒྱུད་གསང་བ་སྙིང་པོའི་སྤྱི་དོན་འོད་གསལ་སྙིང་པོ། ངལ་གསོ་སྐོར་གསུམ། མདོ་ཚོགས་ཆེན་འདུས་པ། ཐེག་ཆེན་ཚུལ་འཇུག བརྒལ་ལན་རྣམ་གསུམ་སོགས་སྔ་འགྱུར་རིང་ལུགས་དྲི་མ་མེད་པའི་མདོ་སྔགས་རྒྱ་མཚོའི་བཀའ་ཁྲིད་གནང་བ་དང་། ཨོ་རྒྱན་མ་ཧཱ་གུ་རུའི་རྣམ་ཐར། བདེན་གཉིས་རྣམ་གཞག བློ་རིགས། རྟགས་རིགས། མཚན་ཉིད་རིགས་པ་བློ་ཡི་ཉིན་བྱེད་སོགས་བསྟན་བཅོས་དུ་མ་ལྗགས་རྩོམ་གནང་། མདོར་ན་སྔོན་ནས་ད་ལྟའི་བར་བཤད་གྲྭ་འདིར་ཐུགས་བརྩེའི་སྒོ་ནས་འཛིན་སྐྱོང་གནང་བ་སོགས་བཀའ་དྲིན་གཞལ་བར་དཀའ་བའོ།། །།

ཕྱིར་ལོག

༄༅། །མཁན་ཆེན་རྣམ་གྲོལ་རིན་པོ་ཆེ་མཆོག་གི་མཛད་རྣམ།

སྤྱི་ལོ་ ༡༩༥༣ ལོར་བོད་སྡེ་དགེར་ཡབ་ཚེ་དྲུང་དང་། ཡུམ་ཆོས་མཚོ་གཉིས་ཀྱི་སྲས་སུ་སྐུ་འཁྲུངས། དགུང་གྲངས་བདུན་པ་ ༡༩༥༩ ལོར་བཙན་བྱོལ་དུ་ཕེབས་ཏེ་མེ་སོར་སྦེལ་ཀོབ་ས་གནས་སུ་བཞུགས། ༡༩༦༨ ལོར་ཟླ་དྲུག་རིང་སྔོན་འགྲོ་འབུམ་ལྔ་བསགས་སྦྱང་མཛད། སྨྱོ་ཤུལ་མཁན་རིན་པོ་ཆེའི་སྐུ་མདུན་ནས་རྒྱལ་སྲས་ལག་ལེན། ཁྱད་པར་འཕགས་བསྟོད། སྟོང་ཐུང་སེངྒེ་ང་རོ། མཁན་པོ་མཁས་གྲུབ་ཀྱི་སྐུ་མདུན་ནས་མི་ཕམ་མཛོད་འགྲེལ་དང་། དབུ་མ་རྒྱན། གྲུབ་བསྡུས། དབུ་མ་འཇུག་པའི་མཆན་འགྲེལ། མཁན་པོ་དཔལ་ལྡན་ཤེས་རབ་ལས་རྩ་ཤེས་མཆན་འགྲེལ་དང་། རིགས་ཚོགས་གཞན་རྣམས་ཀྱི་བཤད་ཁྲིད། དེ་རྗེས་འབྲས་ལྗོངས་སུ་མཁན་པོ་བརྩོན་འགྲུས་ཀྱི་སྐུ་མདུན་དུ་བཅར་ཏེ་ལོ་ལྔའི་རིང་ཤེར་ཕྱིན་གྱི་སྐོར་དང་། དབུ་མའི་སྐོར། ཚད་མའི་སྐོར་སོགས་མདོ་ཕྱོགས་ཐམས་ཅད་དང་། སྔགས་སྐོར་ལས་རང་གྲོལ་སྐོར་གསུམ། གསང་སྙིང་སྤྱི་དོན་མཛོད་ཀྱི་ལྡེ་མིག་བཅས་གསན་བསམ་མཐར་ཕྱིན་མཛད། ཉེར་ལྔ་པ་ ༡༩༧༧ ལོར་ ༸སྐྱབས་རྗེ་རིན་པོ་ཆེའི་བཀའ་དགོངས་ལྟར་མཁན་པོ་བརྩོན་འགྲུས་དང་ལྷན་དགོན་འདིར་ཕེབས། ༡༩༧༨ ལོར་མཐོ་སློབ་དབུ་བརྙེས་ནས་བཟུང་བཤད་གྲྭའི་སྒྲིག་གཞི་གཟིགས་རྟོག་དང་དཔེ་ཁྲིད་སོགས་ཀྱི་འཛིན་སྐྱོང་གནང་། ༡༩༨༣ ལོར་༸སྐྱབས་རྗེ་རིན་པོ་ཆེ་མཆོག་གིས་མཁན་པོའི་མཚན་གནས་བསྩལ། ༡༩༨༨ ལོར་༸མཁན་ཆེན་འཇིགས་མེད་ཕུན་ཚོགས་འབྱུང་གནས་ཀྱི་སྐུ་མདུན་དུ་ཕེབས་ཏེ་མདོ་སྔགས་སྤྱི་དང་། ཁྱད་པར་གསང་སྙིང་སྤྱི་དོན་འོད་གསལ་སྙིང་པོ། ཕྱོགས་བཅུའི་མུན་སེལ། སྤྱི་དོན་མཛོད་ཀྱི་ལྡེ་མིག་བཅས་དང་། རྫོགས་ཆེན་སྐོར་ལས་ལྕེ་བཙུན་སྙིང་ཐིག་དང་མཁན་ཆེན་ཉིད་ཀྱི་རྫོགས་ཆེན་ཁྲིད་ཡིག་སངས་རྒྱས་ལག་སྟེར། ཚིག་གསུམ་གནད་རྡེགས། སེམས་ངོ་མཛུབ་བཙུགས་སོགས་དང་། ༡༩༩༢ ལོར་ཚིག་དོན་རིན་པོ་ཆེའི་མཛོད། མཁའ་འགྲོ་སྙིང་ཐིག་གི་ཁྲིད་ཡིག་ཟབ་ལམ་གསལ་བྱེད། འཇམ་མགོན་མི་ཕམ་རིན་པོ་ཆེའི་ཁྲེགས་ཆོད་གནད་ཀྱི་གདམས་པ་སོགས་གསན། དེ་རྗེས་བཤད་གྲྭ་འདིར་དབུ་མ་འཇུག་པ་དང་། ཤེར་འགྲེལ་ཀེ་ཏ་ཀ།མདོ་སྡེ་རྒྱན། ངེས་ཤེས་སྒྲོན་མེ། སྤྱི་དོན་མཛོད་ཀྱི་ལྡེ་མིག གསང་བདག་དགོངས་རྒྱན་སོགས་སྔ་འགྱུར་རིང་ལུགས་དྲི་མ་མེད་པའི་གཞུང་ལུགས་ཀྱིས་གཙོས། རྒྱ་བོད་ཀྱི་གཞུང་ཆེན་དུ་མའི་བཤད་ཁྲིད་གནང་བ་དང་། བལ་ཡུལ་ཡང་ལེ་ཤོད་དུ་ ༡༩༩༨ ལོར་༸ སྐྱབས་རྗེ་རིན་པོ་ཆེའི་བཀའ་དགོངས་བཞིན་སྒྲུབ་ཁང་བསམ་གཏན་འོད་གསལ་གླིང་རྟེན་བརྟེན་པ་དང་བཅས་གསར་འཛུགས་ཀྱིས་དབང་ཁྲིད་བཀའ་ལུང་སོགས་གནང་དང་གནང་མུས་མཛད་པ་སོགས་མདོར་ན་སྔོན་ནས་ད་ལྟའི་བར་དུ་སངས་རྒྱས་ཀྱི་བསྟན་པ་ཉག་གཅིག་ཐུགས་ལ་བཞག་སྟེ། བཤད་གྲྭ་འདིར་མདོ་སྔགས་ཀྱི་ཆོས་ཤིན་ཏུ་མང་པོ་གསུངས་ཤིང་འཛིན་སྐྱོང་གནང་བ་སོགས་བཀའ་དྲིན་གྱི་ཕུང་པོ་གཞལ་བར་དཀའ་བའོ།། །།

ཕྱིར་ལོག

༄༅། །མཁན་ཆེན་ཚེ་དབང་རྒྱ་མཚོ་རིན་པོ་མཆོག་གི་མཛད་རྣམ།

སྤྱི་ལོ་ ༡༩༥༤ ལོར་བོད་ལྷོ་བྲག་མཁར་ཆུ་ཁུལ་ལྷ་ཁང་རྫོང་དུ་ཡབ་ཤེས་རབ་རྒྱལ་མཚན་དང་། ཡུམ་བསྟན་འཛིན་གཡང་འཛོམས་གཉིས་ཀྱི་སྲས་སུ་སྐུ་འཁྲུངས། འཇམ་དབྱངས་མཁྱེན་བརྩེ་ཆོས་ཀྱི་བློ་གྲོས་མཆོག་གིས་ཚེ་དབང་རྒྱ་མཚོ་ཞེས་པའི་མཚན་བསྩལ་ཅིང་ཕ་མ་གཉིས་ལ་ཕྲུ་གུ་འདི་གཙང་སྦྲ་ཡག་པོ་བྱོས་ཞེས་བཀའ་བསྩལ། ལོ་དེར་ཡབ་ཡུབ་གཉིས་ཀྱིས་ལྷ་ལྡན་ཕྱོགས་སུ་ཕྱག་སྐོར་དུ་ཁྲིད། དགུང་གྲངས་བདུན་པ་ ༡༩༦༠ ལོར་འབྲུག་བརྒྱུད་བཙན་བྱོལ་དུ་ཕེབས། ༡༩༦༢ ལོར་རྡོ་རྗེ་གླིང་དུ་བོད་གཞུང་སློབ་གྲྭར་ཞུགས། བཅུ་གཅིག་ཐོག་ར་སྟོད་སྟོང་ར་སྤྲུལ་སྐུའི་མདུན་ནས་དགེ་ཚུལ་གྱི་སྡོམ་པ་ཞུས་ཤིང་ཆོས་སྒོར་ཞུགས། ༡༩༦༩ ལོར་འཛིན་གྲྭ་མཐར་ཕྱིན། ལོ་དེར་ཝཱར་ཎ་བོད་ཀྱི་ཆེས་མཐོའི་གཙུག་ལག་སློབ་གཉེར་ཁང་དུ་ཞུགས་ཏེ་མཁན་པོ་དཔལ་ལྡན་ཤེས་རབ་སོགས་མཁས་དབང་མང་དུ་བསྟེན། ལོ་དགུའི་རིང་སཾ་སྐྲྀ་ཏ་དང་། དབྱིན་ཇི། ཧིན་དི་སོགས་སྐད་ཡིག་འདྲ་མིན་དང་། ཁྱད་པར་དུ་མདོ་སྔགས་རིག་གནས་དང་བཅས་པར་ཐོས་བསམ་ཧུར་བརྩོན་གནང་། ༡༩༧༨ ལོར་བསླབ་པ་མཐར་ཕྱིན་སྐབས་ཆོས་ལུགས་བཞིའི་རྒྱུགས་སྤྲོད་ཀྱི་རྩེ་གྲས་སུ་སོན་པས།༸སྤྱི་ནོར་༸གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་གིས་ལེགས་སོའི་བསྔགས་བརྗོད་དང་ཐོབ་པར་དཀའ་བའི་དངུལ་གྱི་གཟེངས་རྟགས་བསྩལ། མཁན་ཆེན་ཚེ་དབང་རྒྱ་མཚོ་རིན་པོ་ཆེ། གཞན་ཡང་༸སྐྱབས་རྗེ་མཁྱེན་བརྩེ་རིན་པོ་ཆེའི་མདུན་ནས་སྙིང་ཐིག་ཡ་བཞི།༸སྐྱབས་རྗེ་བདུད་འཇོམས་རིན་པོ་ཆེ་ལས་གཏེར་གསར་སྐོར། ༸སྐྱབས་རྗེ་གྲུབ་དབང་པདྨ་ནོར་བུ་རིན་པོ་ཆེ་ལས་རིན་ཆེན་གཏེར་མཛོད་ཆེན་མོ། གནམ་ཆོས་སྔོན་འགྲོ། རྫོགས་ཆེན་སོགས་དང་། ༸མཁན་ཆེན་འཇིགས་མེད་ཕུན་ཚོགས་མཆོག་དང་སྨྱོ་ཤུལ་མཁན་རིན་པོ་ཆེ་སོགས་མཁས་གྲུབ་རྣམས་ལས་སྔ་འགྱུར་རིང་ལུགས་ཐུན་མིན་ལྟ་གྲུབ་སྐོར་གསན་བསམ་མཐར་ཕྱིན་པར་མཛད། ༡༩༧༩ ལོར་༸སྐྱབས་རྗེ་རིན་པོ་ཆེས་བཀས་བསྐུལ་བ་ལྟར་དགོན་འདིར་ཞབས་སོར་བཀོད་དེ་སློབ་གྲྭའི་འགན་ཁུར་དང་། དགེ་སྐོས་སོགས་ཀྱི་ཐུགས་འགན་ལོ་མང་བཞེས། ༡༩༨༢ ལོར་༸སྐྱབས་རྗེ་གྲུབ་དབང་པདྨ་ནོར་བུ་རིན་པོ་ཆེའི་སྐུ་མདུན་ནས་དགེ་སློང་གི་སྡོམ་པ་བཞེས། ༡༩༨༣ ལོར་༸སྐྱབས་རྗེ་གྲུབ་དབང་པདྨ་ནོར་བུ་རིན་པོ་ཆེ་མཆོག་གིས་མཁན་པོའི་མཚན་གནས་གནང་ཞིང་མངའ་གསོལ་མཛད། དེ་ནས་བཟུང་རྒྱལ་བསྟན་སྤྱི་དང་ཁྱད་པར་སྔ་འགྱུར་བའི་བསྟན་པ་ཁོ་ན་ཐུགས་ལ་བཞག་སྟེ་སྐུ་ངལ་འཛེམས་མེད་ཐོག་ལོ་བཅུ་ཕྲག་ལྷག་རིང་མཐོ་སློབ་འདིར་སུམ་རྟགས་དག་གསུམ་སོགས་ཐུན་མོང་རིག་གནས་དང་། སྤྱོད་འཇུག སྡོམ་གསུམ། རྒྱུད་བླ་མ། ཡིད་བཞིན་མཛོད་འགྲེལ། རྩ་རྒྱུད་གསང་བ་སྙིང་པོ་སོགས་ཐུན་མིན་མདོ་སྔགས་རྒྱ་མཚོའི་བཀའ་ཁྲིད་གནང་། དེ་རྗེས་༸སྐྱབས་རྗེ་རིན་པོ་ཆེའི་དགོངས་བཞེད་བཞིན་ཕྱི་གླིང་ཨ་མི་རི་ཀ་དང་། ཁེ་ན་ཀྲ། ཕི་ལི་པིན། སིང་ག་པུར། ཧོང་ཀོང་སོགས་ཀྱི་ཆོས་ཚོགས་ཁག་ཏུ་ཞབས་སོར་བཀོད་དེ་གཟིགས་རྟོག་གནང་བ་སོགས་མདོར་ན་རྒྱལ་བསྟན་སྤྱི་དང་བྱེ་བྲག་མཐོ་སློབ་འདིར་བཀའ་དྲིན་གྱི་ཕུང་པོ་གཞལ་བར་དཀའོ།། །།

ཕྱིར་ལོག

༄༅། །འཛིན་སྐྱོང་དང་དཔེ་ཁྲིད་གནང་མཁན།

ལྷན་ཚོགས་འདིའི་དགེ་བའི་བཤེས་གཉེན་རྣམས་ནི་བཤད་གྲྭའི་སྔོན་འགྲོའི་འཛིན་གྲྭ་དང་དངོས་གཞི་ལོ་རིམ་དགུའི་དཔེ་ཁྲིད་དང་འཆད་རྩོད་རྩོམ་གསུམ་སོགས་ལ་གཟིགས་རྟོག་གནང་མཁན་ཡིན་ལ། དེ་དག་གིས་སྐྱོར་དཔོན་སྐོས་ཕྱོགས་དང་བཤད་གྲྭའི་ལོ་གཅིག་ནང་བྱ་བ་གལ་ཆེ་བ་མང་པོ་དག་སྤྱི་མོས་སམ་མང་མོས་ཀྱིས་ཐོག་ནས་ཐག་གཅོད་བྱེད་ཀྱི་ཡོད། འདི་འབྲེལ་ལས་གཞིར་ཆེད་གཉེར་གྱི་རྩོམ་སྒྲིག་ཁང་ཟུར་འཛུགས་ཆེད་རྙིང་མའི་སྨོན་ལམ་གཞུང་ནས་མཐུན་རྐྱེན་རོགས་འདེགས་དང་། རྣམ་གྲོལ་གླིང་དགོན་ནས་ཕྱག་ལས་ཀྱི་འགན་བླངས་ཐོག ལམ་སྟོན་པའི་ཐུགས་འགན་མཁན་ཆེན་པདྨ་ཤེས་རབ་མཆོག་ནས་བཞེས་རྒྱུ་དང་། ཕྱོགས་སོ་སོའི་དགོན་སྡེ་རྣམས་ནས་ཀྱང་སྐབས་བབ་ཀྱི་ཕྱག་རོགས་གནང་རྒྱུའི་གཏན་ཁེལ་བྱུང་བ་བཞིན། ཕྱི་ལོ་ ༢༠༡༣ ཟླ་བ་ ༠༤ ཚེས་ ༠༧ ཉིན་འདི་ག་རྣམ་གྲོལ་གླིང་དུ་སྔ་འགྱུར་མཐོ་སློབ་མདོ་སྔགས་རིག་པའི་འབྱུང་གནས་གླིང་གི་འཛིན་སྐྱོང་འོག་ཚིག་མཛོད་རྩོམ་སྒྲིག་པ་བསྐོ་བཞག་གིས་ཚིག་མཛོད་ལས་ཁུངས་སྒོ་འབྱེད་དང་། ཕྱི་ཟླ་ ༠༤ ཚེས་ ༡༠ ནས་དངོས་སུ་ལས་འགོ་ཚུགས་པ་ནས་བཟུང་། གཙོ་བོ་སྔ་འགྱུར་ཐུན་མོང་བའི་ལས་གཞི་རྣམས་ཆེད་ལས་ཐོག ཐུན་མིན་དགོན་སྡེ་སོ་སོ་དང་སྒེར་ནས་བརྒྱུད་བསྐུར་བྱུང་བ་རྣམས་ཀྱང་དག་བཅོས་སྒྲིག་སྦྱོར་ཞུས་ཏེ་ལོ་དགུའི་ལས་འཆར་གྱི་གཏན་འབེབས་ལྟར་ད་བར་ལོ་གསུམ་རེའི་མཚམས་གྱི་ཚིག་མཛོད་ཐོན་རིམ་དང་པོ་དང་གཉིས་པ་ཡན་གྱི་ལས་དོན་གྲུབ་འབྲས་ལེགས་སོན་བྱུང་ཟིན་པ་དང་། ད་ལྟའི་ཆར་འབྱུང་འགྱུར་ཐོན་རིམ་གསུམ་པའི་ཆེད་ལས་དོན་མུ་མཐུད་སྐྱོད་མུས་ཡིན།

ཕྱིར་ལོག

༄༅། །མཁས་དབང་གཞན་གདན་འདྲེན།

༡༩༧༨༠ ལོར་༸སྐྱབས་མགོན་རྒྱལ་བའི་དབང་པོ་མཆོག་གི་ཞབས་ཟུར་དགོན་འདིར་འཁོད། ལོ་དེར་མཁན་རིན་པོ་ཆེ་རིན་ཆེན་མཆོག་ཞབས་སོར་བཀོད་དེ་ལོ་ངོ་གསུམ་རིང་བཤད་གྲྭའི་སློབ་ཕྲུག་ཡོངས་ལ་དམ་པའི་ཆོས་མངོན་པ་མཛོད། དབུ་མ་འཇུག་པ། ཤེར་ཕྱིན་རྣམས་ཀྱི་སློབ་ཁྲིད་གནང་། ༡༩༨༣ ལོར་༸སྐྱབས་རྗེ་དིལ་མགོ་མཁྱེན་བརྩེ་རིན་པོ་ཆེ་མཆོག་ཞབས་སོར་བཀོད་དེ། མཐོ་སློབ་འདིའི་ཕྱག་ཁྱེར་ཀྱི་མཛད་སྒོའི་དབུ་བཞུགས་བཀའ་དྲིན་བསྐྱངས། ལོ་དེར་མཁན་རིན་པོ་ཆེ་ཟླ་ཟེར་མཆོག་ཞབས་སོར་བཀོད་དེ་ལྟ་གྲུབ་ཤན་འབྱེད། བོད་སྤྲུལ་ཤེར་ཕྱིན་འགྲེལ་པ། དོན་རྣམ་ངེས་ཤེས་རབ་རལ་གྲི། བསྟན་དར་ལྷ་རམས་པའི་ཡོན་ཏན་མཛོད་འགྲེལ་གྱི་བཤད་ཁྲིད་དང་། སྔགས་ཕྱོགས་ཀྱི་ལུང་། བསྐྱེད་རིམ་རིམ་ཟིན་བྲིས་ཀྱི་བཤད་ཁྲིད་བཅས་གནང། ༡༩༨༧ ལོར་༸སྐྱབས་རྗེ་དིལ་མགོ་མཁྱེན་བརྩེ་རིན་པོ་ཆེ་མཆོག་ཞབས་སོར་བཀོད་དེ་ཀུན་མཁྱེན་ཀློང་ཆེན་པའི་མཛོད་བདུན། ཀུན་མཁྱེན་རོང་ཟོམ་ཆེན་པོའི་བཀའ་འབུམ། ཀུན་མཁྱེན་མི་ཕམ་ཆོས་རྗེའི་བཀའ་འབུམ་བཅས་ཀྱི་ལྗགས་ལུང་དང་། ཆོས་རྗེ་དཔལ་སྤྲུལ་རིན་པོ་ཆེའི་གསུང་འབུམ་ཕལ་ཆེ་བ་ཁྲིད་གནང་ཞིང་། བསླབ་པ་མཐར་སོན་པ་རྣམས་ལ་ཕྱག་ཁྱེར་གནང་། ༡༩༩༠ ལོར་༸མཁན་ཆེན་འཇིགས་མེད་ཕུན་ཚོགས་འབྱུང་གནས་མཆོག་ཞབས་སོར་བཀོད་དེ། བླ་མའི་ཡང་ཏིག་དང་དེའི་དོན་ཁྲིད། ཚིག་བདུན་རྣམ་བཤད་པདྨ་དཀར་པོ། འཇམ་དཔལ་རྫོགས་པ་ཆེན་པོའི་སྨོན་ལམ། རྩ་གསུམ་སྤྱི་ཡི་བསྙེན་དམིགས། ངེས་ཤེས་རིན་པོ་ཆེའི་སྒྲོན་མེ་བཅས་ཀྱི་བཤད་ཁྲིད་དང་། རྟེན་འབྲེལ་ཉེས་སེལ་ལས་རབ་གླིང་པའི་ཕུར་པ། ཉིད་ཀྱི་གཏེར་ཁ་བཅས་ཀྱི་དབང་གནང་། དེ་སྐབས་ཁོང་ལ་མཁས་གྲུབ་ཡོངས་ཀྱི་དགེ་བའི་བཤེས་གཉེན་ཆེན་པོའི་མཚན་གྱི་ཆེ་བསྟོད་ཨང་རིམ་དང་པོའི་ཕྱག་ཁྱེར་ཕུལ། ༡༩༩༠ ལོ་ནས་བཟུང་ཝཱར་ཎ་བོད་ཀྱི་ཆེས་མཐོའི་གཙུག་ལག་སློབ་གཉེར་ཁང་གི་རིག་གནས་སློབ་དཔོན་ཆེན་པོ་འགོས་པདྨ་རྒྱལ་མཚན་མཆོག་ཞབས་སོར་བཀོད་དེ། ཐུན་མོང་རིག་པའི་གཞུང་ལུགས་ཀྱི་བཤད་ཁྲིད་གནང་བར་བརྟེན་ ༡༩༩༣ ལོར་རིག་གནས་སྨྲ་བའི་དབང་ཕྱུག་ཅེས་པའི་མཚན་གནས་ཕུལ། ལོ་དེར་སྨྱོ་ཤུལ་མཁན་རིན་པོ་ཆེ་མཆོག་ཞབས་སོར་བཀོད་དེ། རྫོགས་པ་ཆེན་པོ་སེམས་ཉིད་ངལ་གསོ། རང་གྲོལ་སྐོར་གསུམ་ཀྱི་དོན་ཁྲིད། ཚིག་གསུམ་གནད་རྡེགས། བློ་སྦྱོང་དོན་བདུན་མའི་བཀའ་ཁྲིད་བཅས་གནང་། ༡༩༩༤ ལོར་༸གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་ཞབས་སོར་འཁོད་སྐབས་སློབ་གཉེར་བ་ཡོངས་ནས་འཆད་རྩོད་རྩོམ་གསུམ་གྱི་རྒྱུགས་ཕུལ། ༡༩༩༤ ལོར་༸སྐྱབས་རྗེ་སྟག་ལུང་རྩེ་སྤྲུལ་རིན་པོ་ཆེ་མཆོག་ཞབས་སོར་བཀོད་དེ་རིན་ཆེན་གཏེར་མཛོད་ཆེན་མོའི་ལྗགས་ལུང་དང་། ཐུགས་རྗེ་ཆེན་པོ་འགྲོ་བ་ཀུན་གྲོོལ་ཀྱི་བཀའ་དབང་བསྩལ། ༡༩༩༦ ལོར་༸སྐྱབས་རྗེ་སྨིན་གླིང་ཁྲི་ཆེན་རིན་པོ་ཆེ་མཆོག་ཞབས་སོར་བཀོད་དེ། རབ་གནས་རྒྱས་པ་དང་ཚེ་དབང་གནང་། ལོ་དེར་ཝཱར་ཎ་རིག་གནས་སློབ་དཔོན་འགོས་པདྨ་རྒྱལ་མཚན་མཆོག་ཕེབས་ཏེ་ཐུན་མོང་རིག་གནས་སྐོར་བཤད་ཁྲིད་གནང་། ༡༩༩༧ ལོར་སྐྱབས་རྗེ་སྟག་ལུང་རྩེ་སྤྲུལ་རིན་པོ་ཆེ་མཆོག་ཞབས་སོར་བཀོད་དེ། རྫོགས་པ་ཆེན་པོ་དགོངས་ཟང་ཐལ་གཙོས་བྱང་གཏེར་ཀྱི་དབང་ལུང་ཆ་ཚང་གནང་། ལོ་དེར་༸གོང་ས་༸སྐྱབས་མགོན་ཆེན་པོ་མཆོག་ཞབས་སོར་བཀའ་དྲིན་བསྐྱངས། ༡༩༩༩ ལོར་མཁན་ཆེན་པདྨ་ཚེ་དབང་མཆོག་ཞབས་སོར་བཀོད་དེ། ལྟ་གྲུབ་ཤན་འབྱེད་ཀྱི་བཤད་ཁྲིད་དང་། བོད་སྤྲུལ་ཤེར་ཕྱིན་ཀྱི་བཤད་ལུང་གནང་ཞིང་། སློབ་ཕྲུག་རྣམས་ཀྱི་དོགས་འདྲི་ལ་ལུང་རིགས་དང་མཐུན་པའི་བཀའ་ལན་གནང་། ལོ་དེའི་ཟླ་ ༤ ལོ་དེར་༸སྤྱི་ནོར་༸གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་ཞབས་སོར་བཀོད་དེ། གཙུག་ལག་ཁང་གསར་པ་ཨོ་རྒྱན་མདོ་སྔགས་བཤད་སྒྲུབ་འོད་གསལ་དར་རྒྱས་གླིང་གི་དབུ་འབྱེད་དང་འབྲེལ། བསླབ་པ་མཐར་སོན་པ་རྣམས་ལ་ཕྱག་ཁྱེར་གནང་ཞིང་། ཕུར་པའི་བཀའ་དབང་དང་ཚེ་དབང་། དེ་བཞིན་བླང་དོར་གྱི་བཀའ་སློབ་གནང་སྟེ་ཉིན་གསུམ་གྱི་རིང་མཛད་ཕྲིན་བཀའ་དྲིན་བསྐྱངས། ༢༠༠༠ ལོར་༸ས་སྐྱ་ཁྲི་འཛིན་སྒྲོལ་མ་ཕོ་བྲང་མཆོག་ཞབས་སོར་བཀོད་ཅིང་རྣམ་གླིང་གཙུག་ལག་ཁང་གསར་པའི་རབ་གནས་རྒྱས་པ་དང་འབྲེལ་ཚེ་དབང་དང་བཀའ་སློབ་ཟབ་རྒྱས་གནང་། ༢༠༠༡ ལོར་༸གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་ཞབས་སོར་བཀོད་དེ་མཁན་ཆེན་ཞི་བ་འཚོའི་སྐོར་འཕགས་བོད་མཁས་པའི་བགྲོ་གླེང་གི་དབུ་འབྱེད་མཛད་སྒོ་དང་འབྲེལ་བཀའ་སློབ་ཟབ་རྒྱས་བསྩལ། ༢༠༠༣ ཟླ ༡ ཚེས་ ༢ ཉིན་༸གོང་ས་སྐྱབས་མགོན་ཆེན་པོ་མཆོག་ཆིབས་བསྒྱུར་བཀའ་དྲིན་ཆེ་བ་བཅས་སོ།། །།

༄༅། །མཁས་པའི་བགྲོ་གླེང་དུ་ཕེབས་པ།

སྤྱི་ལོ་ ༡༩༨༢ ཀེ་ར་ལར་འཛམ་གླིང་ཆོས་ལུགས་ཀྱི་བགྲོ་གླེང་ཐེངས་གཉིས་པའི་ཚོགས་འདུར་མཁན་རིན་པོ་ཆེ་ཚེ་དབང་རྒྱ་མཚོ་དང་། མཁན་རིན་པོ་ཆེ་རིག་འཛིན་རྣམ་གཉིས་ཚོགས་ཕེབས་མཛད། དེ་སྐབས་རྒྱལ་ཁབ་ ༤༥ ནས་ཆོས་ལུགས་ཀྱིིི་མི་སྣ་གལ་ཆེན་ ༣༠༠ ལྷག་ཕེབས་ཡོད། ༡༩༨༨ བོད་རྒྱལ་ལོ་ ༢༡༡༥ ཟླ་ ༡༢ ཚེས་ ༢༧ ཉིན་ཝཱར་ཎ་མཐོ་སློབ་ཏུ་བཙན་བྱོལ་བོད་གཞུང་ཆོས་དོན་ལྷག་ཁག་གིས་གོ་སྒྲིག་གནང་བའི་ཆོས་ཚོགས་ཐེངས་བཞི་པར་རིས་མེད་བོད་ཀྱི་བླ་སྤྲུལ་མཁན་སློབ་རྒྱས་འཛོམས་ཀྱི་ཚོགས་འདུ་ཆེན་མོའི་སྐབས་མཁན་རིན་པོོ་ཆེ་པདྨ་ཤེས་རབ། མཁན་རིན་པོ་ཆེ་ཚེ་དབང་རྒྱ་མཚོ་རྣམ་གཉིས་ཚོགས་ཕེབས་མཛད། ༡༩༩༡ ཟླ་ ༡༠ ཚེས་ ༩ ཉིན་སྦེང་ལོར་དུ་བྱང་ཆུབ་ཆེན་པོའི་ཆོས་ཚོགས་དབུ་བརྙེས་ནས་ལོ་བརྒྱ་འཁོར་བའི་དུས་དྲན་སྐབས། རྒྱལ་སྤྱིའི་ནང་བའི་ཆོས་ལུགས་ཀྱི་ཚོགས་འདུ་ཆེན་མོར་དེང་དུས་འཛམ་གླིང་གི་དཀའ་ངལ་སེལ་ཐབས་སླད་ནང་པའི་ཆོས་ཀྱི་ལྟ་ཚུལ་ཞེས་པའི་བརྗོད་གཞིའི་ཐོག བགྲོ་གླེང་ཚོགས་འདུར་མཁན་རིན་པོོ་ཆེ་པདྨ་ཤེས་རབ། མཁན་པོ་འཇིགས་མེད་སྐལ་བཟང་། སློབ་གཉེར་བ་ཀརྨ་ཕུན་ཚོགས། རྡོ་རྗེ་དབང་ཕྱུག མཁས་གྲུབ་སེངྒེ་བཅས་ཚོགས་ཕེབས་མཛད། ༡༩༩༧ ལོར་དྷ་ས་བོད་གཞུང་སྨན་རྩིས་ཁང་གི་ཚོགས་ཁང་དུ། ཆོས་ཚོགས་ཐེངས་དྲུག་པ་སྤྲུལ་སྐུའི་ཡང་སྲིད་ངོས་འཛིན་གནང་ཕྱོགས་སོགས་ཀྱི་སྐོར་བགྲོ་གླེང་ཚོགས་འདུར་མཁན་རིན་པོ་ཆེ་བཀྲ་ཤིས་ཚེ་རིང་དང་། སློབ་དཔོན་འགྱུར་མེད་རྡོ་རྗེ་རྣམ་གཉིས་ཚོགས་ཕེབས་མཛད། ༡༩༩༨ ལོར་ཧོང་སུར་རྒྱུད་སྨད་གྲྭ་ཚང་དུ་༸གོང་ས་ཆེན་པོ་མཆོག་གིས་དབུས་མཁས་དབང་ཁག་ལྷན་འཛོམས་ཀྱིས་གསང་སྔགས་སྐོར་རྒྱལ་འདུས་མཁས་པའི་བགྲོ་གླེང་ཚོགས་འདུར་མཁན་རིན་པོ་ཆེ་བཀྲ་ཤིས་ཚེ་རིང་ཚོགས་ཕེབས་མཛད། ༡༩༩༨ ལོར་དྷ་ས་ནོར་གླིང་བོད་ཀྱི་རིག་གཞུང་གཅེས་སྐྱོང་ཁང་དུ། ༸གོང་ས་མཆོག་གིས་དབུས་མཁས་དབང་རྣམས་ནས་དགེ་སློང་མའི་སྡོམ་རྒྱུན་སྐོར་བགྲོ་གླེང་སྐབས་མཁན་རིན་པོ་ཆེ་བཀྲ་ཤིས་ཚེ་རིང་དང་། སློབ་དཔོན་འགྱུར་མེད་རྡོ་རྗེ་རྣམ་གཉིས་ཚོགས་ཕེབས་མཛད། ༡༩༩༨ ལོར་སེ་ར་ཐེག་ཆེན་གླིང་ཆོས་སྒར་དུ་ཚན་རིག་སྐོར་བགྲོ་གླེང་ཚོགས་འདུར་མཁན་པོ་རྟ་མགྲིན་སྲི་ཐུབ་ཚོགས་ཕེབས་མཛད། ༡༩༩༩ ལོར་སྦེང་ལོར་ཡེ་ཤུའི་ཆོས་ཚོགས་སུ་རྒྱལ་སྤྱིའི་ཆོས་ལུགས་མཐུན་སྒྲིལ་ཆེད་བགྲོ་གླེང་ཚོགས་འདུར་མཁན་པོ་བསྟན་འཛིན་ནོར་རྒྱས། སྐྱོར་དཔོན་ཉི་མ་དོན་གྲུབ། སློབ་གཉེར་བ་ངེས་དོན་བཟང་པོ། ཕུན་ཚོགས་རྒྱལ་མཚན། བསོད་ནམས་འབུམ་ལྡན། བསོད་ནམས་ཚེ་དབང་། དཀོན་མཆོག་རིག་འཛིན་དང་། བཙུན་དགོན་ནས་བཙུན་མ་ལྷ་མོ་དང་བདེ་སྐྱིད་བཅས་ཚོགས་ཕེབས་མཛད། ༢༠༠༡ ལོར་ཐེག་མཆོག་རྣམ་གྲོལ་བཤད་སྒྲུབ་དར་རྒྱས་གླིང་དུ་མཁན་ཆེན་ཞི་བ་འཚོའི་སྐོར་འཕགས་བོད་མཁས་པའི་བགྲོ་གླེང་ཚོགས་འདུ་ཉིན་གསུམ་རིང་འཚོག་སྐབས། མཁན་རིན་པོ་ཆེ་ཚེ་རིང་རྡོ་རྗེ། མཁན་པོ་ཚེ་དབང་བསོད་ནམས། མཁན་པོ་བསྟན་འཛིན་ནོར་རྒྱས། སློབ་དཔོན་ངག་དབང་བློ་བཟང་། སློབ་དཔོན་མཁྱེན་བརྩེ་ནོར་བུ། སློབ་གཉེར་བ་སྤྲུལ་དགེ་འདུན་རབ་རྒྱས། བསྟན་འཛིན་དབང་ཕྱུག བསོད་ནམས་ཚེ་དབང་བཅས་ཀྱིས་ཚོགས་ཞུགས་མཛད། ༢༠༠༢ ལོར་ལ་དྭགས་དབུས་གཞུང་ནང་པའི་རིག་གནས་ཁང་གི་སློབ་གྲྭར། མཁས་དབང་དྲུག་ཅུ་ལྷག་ལྷན་འཛོམས་ཀྱིས་ཉིན་ལྔའི་རིང་སྡོམ་གསུམ་སྐོར་བགྲོ་གླེང་ཚོགས་འདུར་མཁན་པོ་ཚེ་དབང་བསོད་ནམས་ཚོགས་ཕེབས་མཛད། ༢༠༠༣ ཟླ་ ༡ ཚེས་ ༡༣ ནས་ ༢༥ བར་ཝཱ་རཱ་ཎ་སཱིར་བོད་གཞུང་ཆོས་ཚོགས་ཐེངས་བརྒྱད་པར་མཁན་པོ་ཚེ་དབང་བསོད་ནམས། མཁན་པོ་སངས་རྒྱས་དབང་འདུས། སློབ་དཔོན་པ་སངས་ཚེ་རིང་བཅས་ཚོགས་ཞུགས་མཛད་པ་བཅས་སོ།། །།
